An overview of how my UPPMAX directory is organised.
An image overview of the directory can be found in 'data_organisation.png'.

project: root directory with data_organisation files that overview all files and directories in the project

  data: sub dir to 'project'
    alignments: sub dir to 'data'
      illumina_pacbio: sub dir to 'alignments', output from aligning Illumina reads with the PacBio assembly
    assemblies: sub dir to 'data'
      pacbio_canu_assembly: sub dir to 'assemblies', output files from the Canu assembly of PacBio reads
      pilon_improvement: sub dir to 'assemblies', currently empty, will be output from Pilon improvement of pacbio assembly
    metadata: sub dir to 'data', info about the data
    raw_data: sub dir to 'data'
      illumina: sub dir to 'raw_data', 4 files (soft links) with Illumina reads
      pacbio: sub dir to 'raw_data', 1 file (soft link) with PacBio reads
    transcriptome: sub dir to 'raw_data', files (soft links) with RNA reads

  GenomeAnalysis: sub dir to 'project', local git repository

    analysis: sub dir to 'Genome Analysis'
      alignment: sub dir to 'analysis',
        rna_refseq: sub dir to 'alignment', currently empty, will be .out file from STAR alignment of RNA reads and reference genome
        illumina_pacbio: sub dir to 'alignment', .out file from BWA alignment of Illumina reads and PacBio assembly
      annotation: sub dir to 'analysis'
        functional: sub dir to 'annotation', currently empty, will be output from eggNOGmapper annotation of PacBio assembly
        structural: sub dir to 'annotation', currently empty, will be output from BRAKER annotation of PacBio assembly
      genome_assembly: sub dir to 'analysis'
        canu: sub dir to 'genome_assembly', two .out files from the Canu jobs
      pre_processing: sub dir to 'analysis'
        fastqc: sub dir to 'pre_processing'
          illumina: sub dir to 'fastaqc', output from FastQC of provided Illumina reads
            multiqc_data: sub dir to 'illumina', summary of FastQC results for Illumina reads generated by MultiQC
            zip_files: sub dir to 'illumina', .zip files from FastQC analysis of Illumina reads
          rna: sub dir to 'fastaqc', currently empty, will be output from FastQC of provided RNA reads
            multiqc_data: sub dir to 'rna', currently empty, will be summary of FastQC results for RNA reads generated by MultiQC
            zip_files: sub dir to 'rna', .zip files from FastQC analysis of RNA reads
    code: sub dir to 'GenomeAnalysis', scripts and files with commands

    job_info: sub dir to 'GenomeAnalysis', files with info about submitted jobs (eg graphs over core usage)
